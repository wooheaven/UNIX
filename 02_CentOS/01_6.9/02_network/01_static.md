# Check network
```{bash}
ifconfig -a
service network status
```

# Stop network
```{bash}
service network stop
```

#  Reset all network settings on CentOS
```{bash}
rm /etc/sysconfig/networking/profiles/default/*
rm /etc/sysconfig/networking/devices/*
rm /etc/sysconfig/network-scripts/ifcfg-eth[0-9]
rm /etc/udev/rules.d/70-persistent-net.rules
reboot
```

# static IP ( VirtualBox NAT 기준 )
```{bash}
cat /etc/udev/rules.d/70-persistent-net.rules
# 여기의 address, NAME을 기준으로 ifcfg-eth*를 만든다.

vi /etc/sysconfig/network-scripts/ifcfg-eth0
# static IP 
DEVICE=eth0
ONBOOT=yes
BOOTPROTO=static
IPADDR=10.0.2.16
NETMASK=255.255.255.0
NETWORK=10.0.2.0
    
vi /etc/sysconfig/network
NETWORKING=yes
HOSTNAME=localhost.localdomain
GATEWAY=10.0.2.2
GATEWAYDEV=eth0

vi /etc/resolv.conf
nameserver 168.126.63.1
nameserver 8.8.8.8

vi /etc/hosts
192.168.0.150    server.15
```

# static IP ( eth0 eth1 기준 )
```{bash}
vi /etc/sysconfig/network-scripts/ifcfg-eth0
DEVICE=eth0
TYPE=Ethernet
ONBOOT=yes
BOOTPROTO=static
IPADDR=192.168.7.81
NETMASK=255.255.255.0
GATEWAY=192.168.7.1

vi /etc/sysconfig/network-scripts/ifcfg-eth1
DEVICE=eth1
TYPE=Ethernet
ONBOOT=yes
BOOTPROTO=static
IPADDR=169.168.7.81
NETMASK=255.255.255.0

vi /etc/resolv.conf 
; generated by /sbin/dhclient-script
nameserver 208.91.112.53
nameserver 208.91.112.52
nameserver 219.250.36.130
nameserver 8.8.8.8
```

# restart network
```{bash}
service network restart
```

# check networking
```{bash}
route
or
netstat -r
or
netstat -rn

ping 192.168.0.1 # GATEWAY IP
ping www.google.com
ping 8.8.8.8
```
